#include <stdio.h>
int findUniqueNum(int numbers[], int count) {
    int uniqueNum = 0;
    for (int i = 0; i < count; i++) {
        uniqueNum ^= numbers[i]; // XOR operation
    }
    return uniqueNum;
}
int main() {
    int numCount;
    scanf("%d", &numCount);
    int numList[numCount];
    for (int i = 0; i < numCount; i++) {
        scanf("%d", &numList[i]);
    }
    int answer = findUniqueNum(numList, numCount);
    printf("%d\n", answer);
    return 0;
}
