#include <stdio.h>
void calculateMinMax(int values[5]) {
    long long total = 0;
    long long min_value, max_value;
    for (int i = 0; i < 5; i++) {
        total += values[i];
    }
    min_value = values[0];
    max_value = values[0];
    for (int i = 1; i < 5; i++) {
        if (values[i] < min_value) {
            min_value = values[i];
        }
        if (values[i] > max_value) {
            max_value = values[i];
        }
    }
    long long min_sum_result = total - max_value;
    long long max_sum_result = total - min_value;
    printf("%lld %lld\n", min_sum_result, max_sum_result);
}
int main() {
    int nums[5];
    for (int i = 0; i < 5; i++) {
        scanf("%d", &nums[i]);
    }
    calculateMinMax(nums);
    return 0;
}
