#include <stdio.h>
#include <stdlib.h>
int* countingSort(int input_count, int* input_array, int* output_count) {
    static int frequency_array[100] = {0}; 
    for (int i = 0; i < input_count; i++) {
        int value = input_array[i];
        if (value >= 0 && value < 100) {
            frequency_array[value]++;
        }
    }
    *output_count = 100; // Always return array of size 100
    return frequency_array;
}
int main() {
    int n;
    scanf("%d", &n);
    int* input_array = malloc(sizeof(int) * n);
    for (int i = 0; i < n; i++) {
        scanf("%d", &input_array[i]);
    }
    int output_count;
    int* freq_array = countingSort(n, input_array, &output_count);
    for (int i = 0; i < output_count; i++) {
        printf("%d ", freq_array[i]);
    }
    printf("\n");
    free(input_array);
    return 0;
}
