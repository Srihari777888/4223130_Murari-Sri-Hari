#include <stdio.h>
int isPowerOfTwo(long long n) {
    return (n && !(n & (n - 1)));
}
long long largestPowerOfTwoLessThan(long long n) {
    long long power = 1;
    while (power <= n) {
        power <<= 1;
    }
    return power >> 1;
}
const char* counterGame(long long n) {
    int turn = 0;
    while (n > 0) {
        if (isPowerOfTwo(n)) {
            n /= 2;
        } else {
            n -= largestPowerOfTwoLessThan(n);
        }
        turn ^= 1;
    }
    return turn ? "Richard" : "Louise";
}
int main() {
    int t;
    scanf("%d", &t);
    while (t--) {
        long long n;
        scanf("%lld", &n);
        const char* result = counterGame(n);
        printf("%s\n", result);
    }
    return 0;
}
