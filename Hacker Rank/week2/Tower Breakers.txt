#include <stdio.h>
int towerbreakers(int u, int r) {
    if (r == 1) {
        return 2;
    }
    if (u % 2 == 0) {
        return 2;
    }
    return 1;
}
int main() {
    int T;
    scanf("%d", &T);
    while (T--) {
        int u, r;
        scanf("%d %d", &u, &r);
        int res = towerbreakers(u, r);
        printf("%d\n", res);
    }
    return 0;
}
