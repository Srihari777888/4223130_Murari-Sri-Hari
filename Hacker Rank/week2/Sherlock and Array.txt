#include <stdio.h>
char* balancedSums(int arr[], int n) {
    int totalSum = 0;
    int leftSum = 0;
    for (int i = 0; i < n; i++) {
        totalSum += arr[i];
    }
    for (int i = 0; i < n; i++) {
        totalSum -= arr[i];
        if (leftSum == totalSum) {
            return "YES";
        }
        leftSum += arr[i];
    }
    return "NO";
}
int main() {
    int t;
    scanf("%d", &t);
    while (t--) {
        int n;
        scanf("%d", &n);
        int arr[n];
        for (int i = 0; i < n; i++) {
            scanf("%d", &arr[i]);
        }
        char* result = balancedSums(arr, n);
        printf("%s\n", result);
    }
    return 0;
}
